144inline uint16_t modbusCRC(uint8_t *buf, uint8_t len) {
145 uint16_t crc = 0xFFFF;
146 for (uint8_t i = 0; i < len; i++) {
147 crc ^= buf[i];
148 for (uint8_t j = 0; j < 8; j++) {
149 if (crc & 0x0001) {
150 crc >>= 1;
151 crc ^= 0xA001;

153 crc >>= 1;
154 }
155 }
